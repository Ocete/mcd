\begin{enunciado}
	
	Prueba las siguientes identidades.
	
	\begin{itemize}

		\item $MI(X, Y) = H(X) - H(X|Y) = H(Y) - H(Y|X)$
		\item $MI(X, X) = H(X)$
	\end{itemize}
	
\end{enunciado}
